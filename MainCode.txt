/**
 * @brief Initializes ADC0 for reading from channel 1 (AD0.1).
 */
void ADC0_Init(void) {
    // Configure Pin P0.28 as AD0.1
    PINSEL1 &= 0xFCFFFFFF; // Clear bits 25 and 24
    PINSEL1 |= 0x01000000; // Set bit 24 to 1 (01)

    // Configure the ADC0 Control Register (AD0CR)
    AD0CR = (1 << 1)      |  // SEL = 0x02: Select channel 1 (AD0.1)
            (14 << 8)     |  // CLKDIV: PCLK/15 = 4MHz (assuming 60MHz PCLK)
            (0 << 16)     |  // BURST = 0: Software-controlled conversions
            (1 << 21);       // PDN = 1: Power up the ADC
}


/**
 * @brief Reads a 10-bit value from the configured ADC channel.
 * @return The 10-bit digital result (0-1023).
 */
unsigned int ADC0_Read(void) {
    unsigned int adc_data;
    AD0CR |= (1 << 24); // Start the ADC conversion
    while (!(AD0GDR & (1U << 31))); // Wait for the conversion to complete
    AD0CR &= ~(1 << 24); // Stop the conversion
    adc_data = (AD0GDR >> 6) & 0x3FF; // Extract 10-bit result
    return adc_data;
}


/**
 * @brief Creates a simple software delay.
 * @param ms Delay time in milliseconds.
 */
void delay_ms(unsigned int ms) {
    unsigned int i, j;
    for (i = 0; i < ms; i++) {
        for (j = 0; j < 12000; j++); // Calibrated for approx. 1ms at 60MHz
    }
}
