#include <lpc214x.h>
#include <stdio.h>

void UART0_Init(void) {
    // Configure P0.0 as TXD0 and P0.1 as RXD0
    PINSEL0 |= 0x00000005; 
    
    U0LCR = 0x83; // 8-bit word, 1 stop bit, DLAB = 1
    U0DLL = 97;   // Baud rate 9600 @ 15MHz PCLK
    U0DLM = 0x00;
    U0LCR = 0x03; // DLAB = 0
}

void UART0_TxChar(char ch) {
    while ((U0LSR & 0x20) == 0); // Wait for THRE (Transmit Hold Register Empty)
    U0THR = ch;
}

void UART0_SendString(char* str) {
    while (*str) {
        UART0_TxChar(*str++);
    }
}
